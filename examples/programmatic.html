Example Code

// Initialize SDK manually
import { ThredSDK } from 'thredjs';

const thred = new ThredSDK({
  browserKey: 'your-browser-key',
  debug: true,
  autoInit: false, // Manual init
});

// Initialize
await thred.init();

// Track page view
await thred.trackPageView();

// Identify user
await thred.identify({
  name: 'John Doe',
  email: 'john@example.com',
  company: 'Acme Inc',
});

// Get fingerprint
const fp = thred.getFingerprint();

// Check source
const fromChatGPT = thred.isFromChatGPT();
